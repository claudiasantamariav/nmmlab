    \label{tab:uncoupled_models}
    \begin{tabular}{@{}l  l  l  l@{}}
        \toprule
        \textbf{Model} & \textbf{Polar} & \textbf{Cartesian} & \textbf{Complex} \\ 
        \midrule

        \addlinespace[2ex]
        \textbf{1. Phase‐only harmonic} 
        & $\displaystyle \dot{\theta} = \omega,\quad \dot{r} = 0$
        & $\displaystyle \dot{x} = -\omega\,y,\quad \dot{y} = \omega\,x$
        & $\displaystyle \dot{z} = i\,\omega\,z$
        \\[2ex]

        \addlinespace[2ex]
        \textbf{2. Damped harmonic} 
        & $\displaystyle \dot{r} = \alpha\,r,\quad \dot{\theta} = \omega$
        & $\displaystyle \dot{x} =  \alpha\,x -\omega\,y\quad \dot{y} =  \alpha\,y + \omega\,x$
        & $\displaystyle \dot{z} = (\alpha + i\,\omega)\,z$
        \\[2ex]

        \addlinespace[2ex]
        \textbf{3. Stuart–Landau} 
        & $\displaystyle \dot{r} = \alpha\,r - \gamma\,r^{3},\quad \dot{\theta} = \omega - \beta\,r^{2}$
        & $\displaystyle
          \begin{aligned}
            \dot{x} &= \alpha\,x -\omega\,y 
                    - \gamma\,(x^{2}+y^{2})\,x
                    + \beta\,(x^{2}+y^{2})\,y,\\
            \dot{y} &= \alpha\,y + \omega\,x 
                    - \gamma\,(x^{2}+y^{2})\,y
                    - \beta\,(x^{2}+y^{2})\,x
          \end{aligned}$
        & $\displaystyle \dot{z} = (\alpha + i\,\omega)\,z 
          - (\gamma + i\,\beta)\,\lvert z\rvert^{2}z$
        \\[2ex]

        \addlinespace[2ex]
        \textbf{4. Wilson Cowan}$^{\dagger}$ 
        & --- 
        & $\displaystyle
         \begin{aligned}
          \dot{x} &= -\alpha\,x + S_x\bigl(w_{xx}\,x - w_{xy}\,y + P_x\bigr), \\ 
          \dot{y} &= -\alpha\,y + S_y\bigl(w_{yx}\,x \bigr)
          \end{aligned}$
          
          & --- 
        \\[2ex]

       % \addlinespace[2ex]
       % \textbf{5. Wilson \& Cowan} 
        %& --- 
       % & $\displaystyle
         % \begin{aligned}
           % \tau_x\,\dot{x} &= -\,x + K_x\,S_x\bigl(w_{xx}\,x - w_{xy}\,y + F_x(t)\bigr),\\
           % \tau_y\,\dot{y} &= -\,y + K_y\,S_y\bigl(w_{yx}\,x - %w_{yy}\,y + F_y(t)\bigr)
         % \end{aligned}$
        %& --- 
       % \\[2ex]


        \addlinespace[2ex]
        \textbf{5. Neural Mass Model 1 (NMM1)} \cite{molaee-ardekani_computational_2010}
        & --- 
        & $\displaystyle
          \begin{aligned}
            \frac{1}{\alpha}\,\ddot{x} + 2\,\dot{x}
              &= - \alpha x + S_x\bigl( - w_{xy}\,y + P_x\bigr) \\
            \frac{1}{\alpha}\,\ddot{y} + 2\,\dot{y}
              &= - \alpha y + S_y\bigl(w_{yx}\,x \bigr) 
          \end{aligned}$
        & --- 
        \\[2ex]

        %\addlinespace[2ex]
        %\textbf{7. NMM2} 
        %& --- 
       % & $\displaystyle \dot{x}_1 = f_1(x_1,x_2),\quad \dot{x}_2 = f_2(x_1,x_2)$
        %& --- 
       % \\
       % \bottomrule
    \end{tabular}

    \vspace{1ex}
    \parbox{\textwidth}{\footnotesize
    %\addlinespace[2ex]
    $^{\dagger}$ We assume $\tau_x = \tau_y \equiv \tau$, so that $\alpha = 1/\tau$, and that our sigmoids $S_x, S_y$ include synaptic gain and coupling strength.
    $^{\ddagger}$ External inputs $F_x(t),F_y(t)$ enter inside the sigmoid's $S(\cdot)$ in the Wilson–Cowan models and must exceed a critical threshold to induce a limit cycle (else $F\equiv0$).}

        \textbf{3. Stuart–Landau} & $\displaystyle

          \end{aligned}$ \\[2ex]

       \addlinespace[2ex]
        \textbf{5. Neural Mass Model 1 (NMM1)} \cite{molaee-ardekani_computational_2010} & $\displaystyle
          \begin{aligned}
            \tau_x^{2} \ddot{x} + 2\tau_x \dot{x} + x
              &= K_x S_x\bigl(w_{xx} x - w_{xy} y + P_x + F_x\bigr) ,\\
            \tau_y^{2} \ddot{y} + 2\tau_y \dot{y} + y
              &=  K_y S_y\bigl(w_{yx} x - w_{yy} y \bigr) 
          \end{aligned}$ \\[2ex]

\addlinespace[2ex]
      \textbf{6. NMM2} & $\displaystyle
        \begin{aligned}[t]
        &\tau_x^2\,\ddot x + 2\,\tau_x\,\dot x + x = S_x,\;\;\;
        \tau_y^2\,\ddot y + 2\,\tau_y\,\dot y + y = S_y, \\[6pt]
        &\dot S_x = D_x + 2\,v_x\,S_x,\;\;\;
        \dot S_y = D_y + 2\,v_y\,S_y, \\[6pt]
        &\dot v_x = v_x^2 + w_{xx}\,x - w_{xy}\,y + F_x - (\pi\,S_x)^2, \\[6pt]
        &\dot v_y = v_y^2 + w_{yx}\,x - w_{yy}\,y + F_y - (\pi\,S_y)^2
        \end{aligned}$
